-- Porównaj liczbę wystawionych mandatów i pouczeń w poszczególnych dzielnicach w bieżącym i poprzednich miesiącach.

SELECT 
    (
		-- ostatnie 3 miesiace
        {[Opis Kary].[Rodzaj].&[mandat], [Opis Kary].[Rodzaj].&[pouczenie]} * Tail([Data].[Kalendarz].[Miesiac].Members, 4)
    ) ON COLUMNS,
    [Miejsce].[Dzielnica].[Dzielnica].Members ON ROWS
FROM [policjaHDview]
WHERE ([Measures].[Kary Fakty Count])
